$ python3 pacman.py -p PacmanDQN -n 6000 -x 5000 -l smallGrid

  -p TYPE, --pacman=TYPE
                        the agent TYPE in the pacmanAgents module to use
                        [Default: PacmanDQN]


 -n GAMES, --numGames=GAMES
                        the number of GAMES to play [Default: 6000]


 -x NUMTRAINING, --numTraining=NUMTRAINING
                        How many episodes are training (suppresses output)
                        [Default: 5000]


 -l LAYOUT_FILE, --layout=LAYOUT_FILE
                        the LAYOUT_FILE from which to load the map layout
                        [Default: smallGrid]
